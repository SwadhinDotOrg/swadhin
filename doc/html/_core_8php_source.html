00380     public function findPage($URL){
00381         $pageArr  = explode("/", $URL);
00382         $numSegments = count($pageArr);
00383         
00384         if($numSegments == 1 && $pageArr[0] != "static"){
00385             $this->page = $URL;
00386             $this->functionToCall = DEFAULT_FUNCTION2CALL;
00387         }else{
00388             // Handle static pages first.
00389             if($pageArr[0] == "static"){
00390                 // STATIC: No controller here.
00391                 $this->isStatic = true;
00392                 unset ($pageArr[0]);
00393                 $this->page = implode("/", $pageArr);
00394             }else{
00395                 // DYNAMIC
00396                 // Check if full path exists
00397                 $controllerPath = dirname(__FILE__) . "/../../" . CONTROL_DIR . "/$URL.php";
00398                 if(file_exists($controllerPath)){
00399                     $this->page = $URL;
00400                     $this->functionToCall = DEFAULT_FUNCTION2CALL;
00401                 }else{
00402                     // Check later.
00403                     $this->functionToCall = $pageArr[$numSegments - 1];
00404                     unset($pageArr[$numSegments - 1]);
00405                     $this->page = implode("/", $pageArr);
